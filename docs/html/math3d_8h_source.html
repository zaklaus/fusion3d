858  Quaternion(const Matrix4f& m)
859  {
860  float trace = m[0][0] + m[1][1] + m[2][2];
861 
862  if(trace > 0)
863  {
864  float s = 0.5f / sqrtf(trace + 1.0f);
865  (*this)[3] = 0.25f / s;
866  (*this)[0] = (m[1][2] - m[2][1]) * s;
867  (*this)[1] = (m[2][0] - m[0][2]) * s;
868  (*this)[2] = (m[0][1] - m[1][0]) * s;
869  }
870  else if(m[0][0] > m[1][1] && m[0][0] > m[2][2])
871  {
872  float s = 2.0f * sqrtf(1.0f + m[0][0] - m[1][1] - m[2][2]);
873  (*this)[3] = (m[1][2] - m[2][1]) / s;
874  (*this)[0] = 0.25f * s;
875  (*this)[1] = (m[1][0] + m[0][1]) / s;
876  (*this)[2] = (m[2][0] + m[0][2]) / s;
877  }
878  else if(m[1][1] > m[2][2])
879  {
880  float s = 2.0f * sqrtf(1.0f + m[1][1] - m[0][0] - m[2][2]);
881  (*this)[3] = (m[2][0] - m[0][2]) / s;
882  (*this)[0] = (m[1][0] + m[0][1]) / s;
883  (*this)[1] = 0.25f * s;
884  (*this)[2] = (m[2][1] + m[1][2]) / s;
885  }
886  else
887  {
888  float s = 2.0f * sqrtf(1.0f + m[2][2] - m[1][1] - m[0][0]);
889  (*this)[3] = (m[0][1] - m[1][0]) / s;
890  (*this)[0] = (m[2][0] + m[0][2]) / s;
891  (*this)[1] = (m[1][2] + m[2][1]) / s;
892  (*this)[2] = 0.25f * s;

915  {
916  static const float EPSILON = 1e3;
917 
918  float cos = this->Dot(r);
919  Quaternion correctedDest;
920 
921  if(shortestPath && cos < 0)
922  {
923  cos *= -1;
924  correctedDest = r * -1;
925  }
926  else
927  correctedDest = r;

930  return NLerp(correctedDest, lerpFactor, false);

933  float angle = atan2(sin, cos);
