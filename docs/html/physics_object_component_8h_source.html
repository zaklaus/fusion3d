70  virtual void DataDeploy(tinyxml2::XMLElement* data)
71  {

77  if (data->Attribute("shape"))

80 
81  if (s == "bvh")
82  {
83  shape = new btBvhTriangleMeshShape(Mesh::ImportCollision(data->GetText()), true);
84  }
85  else if (s == "convex")
86  {
87  shape = new btConvexTriangleMeshShape(Mesh::ImportCollision(data->GetText()), true);
88  }
89  else if (s == "box")
90  {
91  shape = new btBoxShape(Util::ParseVector3(data->GetText()).GetBT());
92  }
93  else if (s == "sphere")
94  {
95  shape = new btSphereShape(atof(data->GetText()));
96  }
97  else if (s == "capsule")
98  {
99  std::vector<std::string> p = Util::Split(data->GetText(), ';');
100  shape = new btCapsuleShape(atof(p[0].c_str()), atof(p[1].c_str()));
101  }
102  else if (s == "cylinder")
103  {
104  shape = new btCylinderShape(Util::ParseVector3(data->GetText()).GetBT());
105  }
106  else
107  {
108  std::cerr << "Unsupported collision shape! " << s << std::endl;
109  assert(1 == 0);
110  }
111  }
112  else
113  {
114  std::cerr << "Unspecified collision shape! " << std::endl;
115  assert(1 == 0);
116  }
117 
118  if (data->Attribute("mass"))
119  {
120  mass = atof(data->Attribute("mass"));
121  }
122 
123  if (data->Attribute("calcInertia"))
124  {
125  calcInertia = (!strcmp(data->Attribute("calcInertia"), "0")) ? false : true;
126  }
127 
128  if (data->Attribute("inertia"))

130  inertia = Util::ParseVector3(data->Attribute("inertia"));

136  }
