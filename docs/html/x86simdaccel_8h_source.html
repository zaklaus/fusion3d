295  inline SIMD4f Round(int roundingMode = 0) const
296  {
297  roundingMode = roundingMode & 3; //Use low bits, since rounding mode only uses 2 bits.
298 
299  #if SIMD_SUPPORTED_LEVEL >= SIMD_LEVEL_x86_SSE4_1
300  return _mm_round_ps(m_data, roundingMode);
301  #else
302  //If the hardware doesn't support rounding, it must be emulated
303  //by manipulating the MXCSR modes.
304  uint32_t mxcsr = _mm_getcsr();
305  uint32_t newmxcsr = mxcsr | (roundingMode << 13);
306  _mm_setcsr(newmxcsr);
307 
308  static const __m128 SIGN_MASK = _mm_castsi128_ps(SIMD4i(0x80000000,0x80000000,0x80000000,0x80000000));
309  static const __m128 MAGIC_ROUNDING_CONSTANT = _mm_castsi128_ps(SIMD4i(0x4B000000,0x4B000000,0x4B000000,0x4B000000));
310  SIMD4f sign = _mm_and_ps(m_data, SIGN_MASK);
311  SIMD4f signedMagic = _mm_or_ps(MAGIC_ROUNDING_CONSTANT, sign);
312  SIMD4f dataPlusSignedMagic = _mm_add_ps(m_data, signedMagic);
313 
314  SIMD4f result = SIMD4f(_mm_sub_ps(dataPlusSignedMagic, signedMagic));
315 
316  _mm_setcsr(mxcsr);
317  return result;
318  #endif
319  }
